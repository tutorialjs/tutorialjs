Tutorial.js

A lightweight introductory tool for your application

How to

<link rel="stylesheet" href="tutorial.css" />
<script src="tutorial.js"></script>

Initialization

var guide = new Tutorial({
    steps: [
        {
            highlight: ".myclass1",
            text: "my first text"
        },
        {
            highlight: ".mysecondclass",
            text: "my second text"
        }],
    buttons: {next: "Weiter"},
    autoplay: true
)};
